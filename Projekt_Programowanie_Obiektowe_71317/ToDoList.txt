Do klasy Asset walidacja nazwy (nie może być pusta i nie może przekraczać 100 znaków).
private string _name;

        public string Name
        {
            get => _name;
            set
            {
                // 1. Sprawdzamy czy nie jest puste
                if (string.IsNullOrWhiteSpace(value))
                {
                    throw new ArgumentException("Imię nie może być puste!");
                }

                // 2. Sprawdzamy długość
                if (value.Length < 2)
                {
                    throw new ArgumentException("Imię jest za krótkie!");
                }

                /*
                   Przez to że z tej klasy dziedziczą inne klasy, np Vehicle a ono może mieć w nazwie cyfry,
                // 3. Sprawdzamy czy nie zawiera cyfr
                if (value.Any(char.IsDigit))
                {
                    throw new ArgumentException("Imię nie może zawierać cyfr!");
                }
                */

                // Jeśli wszystko OK, przypisujemy (z dużej litery dla porządku)
                _name = char.ToUpper(value[0]) + value.Substring(1).ToLower();
            }
        }
